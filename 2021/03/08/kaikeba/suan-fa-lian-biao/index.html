var hasCycle = function(head) {
    let slow = head;
    let fast = head;
    while (fast && fast.next) {
        slow = slow.next;
        fast = fast.next.next;
        if (fast == slow) {
            return true;
        }
    }
    return false;
};

var detectCycle = function(head) {
    if (!head) return null;
    let pre = head,
        cur = head;
    while (cur && cur.next) {
        pre = pre.next;
        cur = cur.next.next;
        if (pre === cur) {
            let temp = head;
            while (pre !== temp) {
                pre = pre.next;
                temp = temp.next;
            }
            return pre;
        }
    }
    return null;
};

var isHappy = function(n) {
    let pre = n,
        cur = getNext(n);
    while (cur != pre && cur != 1) {
        pre = getNext(pre);
        cur = getNext(getNext(cur));
    }
    return cur === 1;
};

var getNext = function(n) {
    let t = 0;
    while (n) {
        t += (n % 10) * (n % 10)
        n = Math.floor(n / 10);
    }
    return t;
}

var rotateRight = function(head, k) {
    //先判断首节点
    if (!head) return null;
    let cur = head,
        size = 1;
    //找到尾节点，穿成环
    while (cur.next) {
        cur = cur.next;
        size += 1;
    };
    cur.next = head;
    //获取到链表的长度，找到第size-k个节点，然后断开
    for (let i = 1; i < size - (k % size); i++) {
        head = head.next;
    }
    cur = head.next;
    head.next = null;
    return cur;
};

24 两两交换链表中的节点

var swapPairs = function(head) {
    //判断首节点是否为空
    if (!head) return null;
    //将相邻的两个链表进行反转
    let ret = new ListNode(-1, head),
        temp = ret;
    while (temp.next && temp.next.next) {
        let pre = temp.next,
            cur = temp.next.next;
        pre.next = cur.next;
        cur.next = pre;
        temp.next = cur;
        temp = pre;
    }
    return ret.next;
};

var removeNthFromEnd = function(head, n) {
    if(!head) return null;
    //考虑正常情况
    var p = q = r = s = head;
    for(let i=1;i<n;i++){
        if(p.next){
            p = p.next;
        }else{
            return q;
        }
    }
    while(p.next){
        if(q==r){
            p = p.next;
            q = q.next;
        }else{
            p=p.next;
            q=q.next;
            r=r.next;
        }
    }
    if(r==q ){
        return s.next;
    }
    if(p==q){
        r.next =null;
        return s;
    }
    r.next =q.next;
    return s;
};

参考解析

var removeNthFromEnd = function(head, n) {
    if (!head) return null;
    //创建一个虚头，两个指针
    //一个指向虚头，一个指向真是的头节点cur
    let ret = new ListNode(-1, head),
        pre = ret,
        cur = head;
    //cur移动k步
    for (let i = 0; i < n; i++) {
        cur = cur.next;
    }
    if (!cur) return head.next;
    //然后两个指针仪器移动，知道cur指向空
    while (cur) cur = cur.next, pre = pre.next;
    //然后进行删除操作
    pre.next = pre.next.next;
    return ret.next;
};

2] 两数相加

var addTwoNumbers = function(l1, l2) {
    let p = l1,
        q = l2;
    let result, re, r;
    var s = 0;

    while (p || q) {
        let a = p != null ? p.val : 0,
            b = q != null ? q.val : 0;
        //余数
        let d = (a + b + s) % 10;
        r = new ListNode(d);
        if (!result) {
            result = r;
            re = r;
        } else {
            re.next = r;
            re = re.next;
        }
        //整数
        s = Math.floor((a + b + s) / 10);
        p = p != null ? p.next : null;
        q = q != null ? q.next : null;
    }
    if (s > 0) {
        r = new ListNode(s);
        r.next = null;
        re.next = r;
    }
    return result;
};

622. 设计循环队列

var MyCircularQueue = function(k) {
    this.queue = Array(k + 1);
    this.front = 0;
    this.rear = 0;
    this.max = k;
};

/** 
 * @param {number} value
 * @return {boolean}
 */
MyCircularQueue.prototype.enQueue = function(value) {
    if (this.isFull()) return false;
    this.queue[this.rear] = value;
    this.rear = (this.rear + 1) % (this.max + 1);
    return true;
};

/**
 * @return {boolean}
 */
MyCircularQueue.prototype.deQueue = function() {
    if (this.isEmpty()) return false;
    this.front = (this.front + 1) % (this.max + 1);
    return true;
};

/**
 * @return {number}
 */
MyCircularQueue.prototype.Front = function() {
    if (this.isEmpty()) return -1;
    return this.queue[this.front];
};

/**
 * @return {number}
 */
MyCircularQueue.prototype.Rear = function() {
    if (this.isEmpty()) return -1;
    return this.queue[(this.rear + this.max) % (this.max + 1)];
};

/**
 * @return {boolean}
 */
MyCircularQueue.prototype.isEmpty = function() {
    return ((this.rear - this.front + this.max + 1) % (this.max + 1)) == 0;
};

/**
 * @return {boolean}
 */
MyCircularQueue.prototype.isFull = function() {
    return (this.rear - this.front + this.max + 1) % (this.max + 1) == this.max;
};
